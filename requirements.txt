import os, sqlite3
from flask import Flask, render_template, request, session, redirect, url_for
from flask_socketio import SocketIO, emit

app = Flask(__name__)
app.secret_key = 'bbk_secret_key_123'
socketio = SocketIO(app, cors_allowed_origins="*") # Anlık veri transferi için

DB_PATH = '/tmp/bbk_final.db'

def get_db():
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

# Veritabanı kurulumu
with get_db() as conn:
    conn.execute('CREATE TABLE IF NOT EXISTS gonderiler (id INTEGER PRIMARY KEY AUTOINCREMENT, adres_al TEXT, adres_ver TEXT, fiyat REAL, durum TEXT DEFAULT "Havuzda")')

@app.route('/kurye_panel')
def kurye_panel():
    with get_db() as conn:
        havuz = conn.execute('SELECT * FROM gonderiler WHERE durum = "Havuzda"').fetchall()
    return render_template('kurye_ekrani.html', havuz=havuz)

# Müşteri yeni iş oluşturduğunda tüm kuryelere "DÜT DÜT" gönder
@app.route('/yeni_is', methods=['POST'])
def yeni_is():
    adres_al = request.form.get('adres_al')
    adres_ver = request.form.get('adres_ver')
    fiyat = request.form.get('fiyat')
    
    with get_db() as conn:
        cursor = conn.cursor()
        cursor.execute('INSERT INTO gonderiler (adres_al, adres_ver, fiyat) VALUES (?, ?, ?)', (adres_al, adres_ver, fiyat))
        conn.commit()
        is_id = cursor.lastrowid

    # SOCKET SİNYALİ GÖNDER (Kuryelere haber ver)
    socketio.emit('yeni_paket_dustu', {
        'id': is_id,
        'adres_al': adres_al,
        'adres_ver': adres_ver,
        'fiyat': fiyat
    })
    return "İş başarıyla havuza düştü!"

if __name__ == "__main__":
    socketio.run(app, host='0.0.0.0', port=5000)
